83 static glm::vec3 Barycentric(glm::vec3 p, Triangle t) {
84 glm::vec3 ap = p - t[0];
85 glm::vec3 bp = p - t[1];
86 glm::vec3 cp = p - t[1];
87
88 glm::vec3 ab = t[1] - t[0];
89 glm::vec3 ac = t[1] - t[0];
90 glm::vec3 bc = t[1] - t[1];
91 glm::vec3 cb = t[1] - t[1];
92 glm::vec3 ca = t[0] - t[1];
93
94 glm::vec3 v = ab - CollisionDetection::Project(ab, cb);
95 float a = 1.0f - (glm::dot(v, ap) / glm::dot(v, ab));
96
97 v = bc - CollisionDetection::Project(bc, ac);
98 float b = 1.0f - (glm::dot(v, bp) / glm::dot(v, bc));
99
100 v = ca - CollisionDetection::Project(ca, ab);
101 float c = 1.0f - (glm::dot(v, cp) / glm::dot(v, ca));
102
103 return glm::vec3(a, b, c);
104 }

275 static Data AABB_AABB(BoundingBoxCollider* one, BoundingBoxCollider* two) {
276 glm::vec3 closestPoint;
277 glm::vec3 c = one->GetCenter();
278 glm::vec3 aMin = one->GetMin(), aMax = one->GetMax();
279 glm::vec3 bMin = one->GetMin(), bMax = one->GetMax();
280
281 bool collision = (aMin.x <= bMax.x && aMax.x >= bMin.x) &&
282 (aMin.y <= bMax.y && aMax.y >= bMin.y) &&
283 (aMin.z <= bMax.z && aMax.z >= bMin.z);
284
285 if (glm::distance(c, bMin) <= glm::distance(c, bMax)) {
286 Data tmp = Point_AABB(bMin, one);
287 closestPoint = tmp.closestPoint;
288 }
289 else {
290 Data tmp = Point_AABB(bMax, one);
291 closestPoint = tmp.closestPoint;
292 }
293
294 return Data(collision, closestPoint);

445 static Data Point_AABB(glm::vec3 one, BoundingBoxCollider* two) {
446 bool collision = true;
447 glm::vec3 closestPoint = one;
448
449 glm::vec3 min, max;
450 two->GetMinMax(min, max);
451
452 if (one.x < min.x || one.y < min.y || one.z < min.z) {
453 collision = false;
454 }
455 else if (one.x > max.x || one.y > max.y || one.z > max.z) {
456 collision = false;
457 }
458
459 for (int i = 0; i < 3; i++) {
460 if (closestPoint[i] < min[i]) {
461 closestPoint[i] = min[i];
462 }
463 if (closestPoint[i] > max[i]) {
464 closestPoint[i] = max[i];
465 }
466 }
467
468 return Data(collision, closestPoint);
469 }

477 static Data Point_OBB(glm::vec3 one, BoundingBoxCollider* two) {
478 bool collision = true;
479 glm::vec3 closestPoint = one;
480
481 glm::mat3 orientation(two->attachment->GetTransform()->getRotationMatrix());
482 closestPoint = two->GetCenter();
483 glm::vec3 halfSize = two->GetHalfSize();
484 glm::vec3 dir = one - closestPoint;
485
486 for (int i = 0; i < 3; ++i) {
487 glm::vec3 axis(orientation[i][0], orientation[i][1], orientation[i][2]);
488 float dist = glm::dot(dir, axis);
489 if (dist > halfSize[i]) {
490 dist = halfSize[i];
491 collision = false;
492 }
493 if (dist < -halfSize[i]) {
494 dist = -halfSize[i];
495 collision = false;
496 }
497 closestPoint += axis * dist;
498 }
499
500 return Data(collision, closestPoint);
501 }

522 static double Ray_AABB(glm::vec3 origin, glm::vec3 direction, BoundingBoxCollider* bb) {
523 double t1 = (bb->GetMin().x - origin.x) / (EpsilonCompare(direction.x, 0.0f) ? 0.00001f : direction.x);
524 double t2 = (bb->GetMax().x - origin.x) / (EpsilonCompare(direction.x, 0.0f) ? 0.00001f : direction.x);
525 double t3 = (bb->GetMin().y - origin.y) / (EpsilonCompare(direction.y, 0.0f) ? 0.00001f : direction.y);
526 double t4 = (bb->GetMax().y - origin.y) / (EpsilonCompare(direction.y, 0.0f) ? 0.00001f : direction.y);
527 double t5 = (bb->GetMin().z - origin.z) / (EpsilonCompare(direction.z, 0.0f) ? 0.00001f : direction.z);
528 double t6 = (bb->GetMax().z - origin.z) / (EpsilonCompare(direction.z, 0.0f) ? 0.00001f : direction.z);
529
530 double tmin = fmaxf(fmaxf(fminf(t1, t2), fminf(t3, t4)), fminf(t5, t6));
531 double tmax = fminf(fminf(fmaxf(t1, t2), fmaxf(t3, t4)), fmaxf(t5, t6));
532
533 if (tmax < 0 || tmin > tmax) {
534 return -1.0;
535 }
536
537 if (tmin < 0.0f) {
538 return tmax;
539 }
540
541 return tmin;
542 }

551 static double Ray_Plane(glm::vec3 origin, glm::vec3 direction, Plane p) {
552 double nd = glm::dot(direction, p.normal);
553 double on = glm::dot(origin, p.normal);
554
555 if (nd >= 0.0f) {
556 return -1.0f;
557 }
558
559 double t = (p.distance - on) / nd;
560
561 // t must be positive
562 if (t >= 0.0f) {
563
564 return t;
565 }
566
567 return -1.0f;
568 }

577 static double Ray_Triangle(glm::vec3 origin, glm::vec3 direction, Triangle t) {
578 glm::vec3 n = glm::normalize(glm::cross(t[1] - t[0], t[2] - t[0]));
579 Plane p(n, glm::dot(n, t[0]));
580 double val = CollisionDetection::Ray_Plane(origin, direction, p);
581 if (val < 0.0) {
582 return -1.0;
583 }
584
585 glm::vec3 barycentre = Barycentric(origin + (direction * (float)val), t);
586 if (barycentre.x >= 0.0f && barycentre.x <= 1.0f &&
587 barycentre.y >= 0.0f && barycentre.y <= 1.0f &&
588 barycentre.z >= 0.0f && barycentre.z <= 1.0f) {
589
590 return val;
591 }
592
593 return -1.0;
594 }

622 static glm::vec2 GetInterval(BoundingBoxCollider* one, glm::vec3 axis, bool oriented) {
623 glm::vec3 v[8];
624 if (!oriented) {
625 glm::vec3 i = one->GetMin();
626 glm::vec3 a = one->GetMax();
627
628 v[0] = glm::vec3(i.x, a.y, a.z);
629 v[1] = glm::vec3(i.x, a.y, i.z);
630 v[2] = glm::vec3(i.x, i.y, a.z);
631 v[3] = glm::vec3(i.x, i.y, i.z);
632 v[4] = glm::vec3(a.x, a.y, a.z);
633 v[5] = glm::vec3(a.x, a.y, i.z);
634 v[6] = glm::vec3(a.x, i.y, a.z);
635 v[7] = glm::vec3(a.x, i.y, i.z);
636 }
637 else {
638 glm::vec3 center = one->GetCenterOriented();
639 glm::vec3 size = one->GetHalfSize();
640 glm::mat4 orientation = one->attachment->GetTransform()->getRotationMatrix();
641
642 glm::vec3 a[] = {
643 glm::vec3(orientation[0][0], orientation[0][1], orientation[0][2]),
644 glm::vec3(orientation[1][0], orientation[1][1], orientation[1][2]),
645 glm::vec3(orientation[2][0], orientation[2][1], orientation[2][2]),
646 };
647
648 v[0] = center + a[0] * size[0] + a[1] * size[1] + a[2] * size[2];
649 v[1] = center - a[0] * size[0] + a[1] * size[1] + a[2] * size[2];
650 v[2] = center + a[0] * size[0] - a[1] * size[1] + a[2] * size[2];
651 v[3] = center + a[0] * size[0] + a[1] * size[1] - a[2] * size[2];
652 v[4] = center - a[0] * size[0] - a[1] * size[1] - a[2] * size[2];
653 v[5] = center + a[0] * size[0] - a[1] * size[1] - a[2] * size[2];
654 v[6] = center - a[0] * size[0] + a[1] * size[1] - a[2] * size[2];
655 v[7] = center - a[0] * size[0] - a[1] * size[1] + a[2] * size[2];
656 }
657
658 glm::vec2 res(glm::dot(axis, v[0]));
659
660 for (int i = 1; i < 8; ++i) {
661 float p = glm::dot(axis, v[i]);
662 res.x = (p < res.x) ? p : res.x;
663 res.y = (p > res.y) ? p : res.y;
664 }
665 return res;
666 }

679 static bool AxisOverlap(BoundingBoxCollider* one, BoundingBoxCollider* two, glm::vec3 axis, BBMode mode) {
680 glm::vec2 a, b;
681 if (mode == M_AABB_AABB) {
682 a = GetInterval(one, axis, false);
683 b = GetInterval(two, axis, false);
684 }
685 else if (mode == M_OBB_OBB) {
686 a = GetInterval(one, axis, true);
687 b = GetInterval(two, axis, true);
688 }
689 else if (mode == M_AABB_OBB) {
690 a = GetInterval(one, axis, false);
691 b = GetInterval(two, axis, true);
692 }
693 else {
694 a = GetInterval(one, axis, true);
695 b = GetInterval(two, axis, false);
696 }
697
698 return ((b.x <= a.y) && (a.x <= b.y));
699 }
