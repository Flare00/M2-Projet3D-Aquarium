235 Data AABB_AABB(BoundingBoxCollider* one, BoundingBoxCollider* two) {
236 glm::vec3 closestPoint;
237 glm::vec3 c = one->GetCenter();
238 glm::vec3 aMin = one->GetMin(), aMax = one->GetMax();
239 glm::vec3 bMin = one->GetMin(), bMax = one->GetMax();
240
241 bool collision = (aMin.x <= bMax.x && aMax.x >= bMin.x) &&
242 (aMin.y <= bMax.y && aMax.y >= bMin.y) &&
243 (aMin.z <= bMax.z && aMax.z >= bMin.z);
244
245 if (glm::distance(c, bMin) <= glm::distance(c, bMax)) {
246 Data tmp = Point_AABB(bMin, one);
247 closestPoint = tmp.closestPoint;
248 }
249 else {
250 Data tmp = Point_AABB(bMax, one);
251 closestPoint = tmp.closestPoint;
252 }
253
254 return Data(collision, closestPoint);
255 }

270 glm::vec3 test[15] = {

405 Data Point_AABB(glm::vec3 one, BoundingBoxCollider* two) {
406 bool collision = true;
407 glm::vec3 closestPoint = one;
408
409 glm::vec3 min = two->GetMin(), max = two->GetMax();
410 if (one.x < min.x || one.y < min.y || one.z < min.z) {
411 collision = false;
412 }
413 else if (one.x > max.x || one.y > max.y || one.z > max.z) {
414 collision = false;
415 }
416
417 for (int i = 0; i < 3; i++) {
418 if (closestPoint[i] < min[i]) {
419 closestPoint[i] = min[i];
420 }
421 if (closestPoint[i] > max[i]) {
422 closestPoint[i] = max[i];
423 }
424 }
425
426 return Data(collision, closestPoint);

435 Data Point_OBB(glm::vec3 one, BoundingBoxCollider* two) {
436 bool collision = true;
437 glm::vec3 closestPoint = one;
438
439 glm::mat3 orientation(two->attachment->GetTransform()->getRotationMatrix());
440 closestPoint = two->GetCenter();
441 glm::vec3 halfSize = two->GetHalfSize();
442 glm::vec3 dir = one - closestPoint;
443
444 for (int i = 0; i < 3; ++i) {
445 glm::vec3 axis(orientation[i][0], orientation[i][1], orientation[i][2]);
446 float dist = glm::dot(dir, axis);
447 if (dist > halfSize[i]) {
448 dist = halfSize[i];
449 collision = false;
450 }
451 if (dist < -halfSize[i]) {
452 dist = -halfSize[i];
453 collision = false;
454 }
455 closestPoint += axis * dist;
456 }
457
458 return Data(collision, closestPoint);
459 }

498 glm::vec2 GetInterval(BoundingBoxCollider* one, glm::vec3 axis, bool oriented) {
499 glm::vec3 v[8];
500 if (!oriented) {
501 glm::vec3 i = one->GetMin();
502 glm::vec3 a = one->GetMax();
503
504 v[0] = glm::vec3(i.x, a.y, a.z);
505 v[1] = glm::vec3(i.x, a.y, i.z);
506 v[2] = glm::vec3(i.x, i.y, a.z);
507 v[3] = glm::vec3(i.x, i.y, i.z);
508 v[4] = glm::vec3(a.x, a.y, a.z);
509 v[5] = glm::vec3(a.x, a.y, i.z);
510 v[6] = glm::vec3(a.x, i.y, a.z);
511 v[7] = glm::vec3(a.x, i.y, i.z);
512 }
513 else {
514 glm::vec3 center = one->GetCenterOriented();
515 glm::vec3 size = one->GetHalfSize();
516 glm::mat4 orientation = one->attachment->GetTransform()->getRotationMatrix();
517
518 glm::vec3 a[] = {
519 glm::vec3(orientation[0][0], orientation[0][1], orientation[0][2]),
520 glm::vec3(orientation[1][0], orientation[1][1], orientation[1][2]),
521 glm::vec3(orientation[2][0], orientation[2][1], orientation[2][2]),
522 };
523
524 v[0] = center + a[0] * size[0] + a[1] * size[1] + a[2] * size[2];
525 v[1] = center - a[0] * size[0] + a[1] * size[1] + a[2] * size[2];
526 v[2] = center + a[0] * size[0] - a[1] * size[1] + a[2] * size[2];
527 v[3] = center + a[0] * size[0] + a[1] * size[1] - a[2] * size[2];
528 v[4] = center - a[0] * size[0] - a[1] * size[1] - a[2] * size[2];
529 v[5] = center + a[0] * size[0] - a[1] * size[1] - a[2] * size[2];
530 v[6] = center - a[0] * size[0] + a[1] * size[1] - a[2] * size[2];
531 v[7] = center - a[0] * size[0] - a[1] * size[1] + a[2] * size[2];
532 }
533
534 glm::vec2 res(glm::dot(axis, v[0]));
535
536 for (int i = 1; i < 8; ++i) {
537 float p = glm::dot(axis, v[i]);
538 res.x = (p < res.x) ? p : res.x;
539 res.y = (p > res.y) ? p : res.y;
540 }
541 return res;
542 }

555 bool AxisOverlap(BoundingBoxCollider* one, BoundingBoxCollider* two, glm::vec3 axis, BBMode mode) {
556 glm::vec2 a, b;
557 if (mode == M_AABB_AABB) {
558 a = GetInterval(one, axis, false);
559 b = GetInterval(two, axis, false);
560 }
561 else if (mode == M_OBB_OBB) {
562 a = GetInterval(one, axis, true);
563 b = GetInterval(two, axis, true);
564 }
565 else if (mode == M_AABB_OBB) {
566 a = GetInterval(one, axis, false);
567 b = GetInterval(two, axis, true);
568 }
569 else {
570 a = GetInterval(one, axis, true);
571 b = GetInterval(two, axis, false);
572 }
573
574 return ((b.x <= a.y) && (a.x <= b.y));
575 }
