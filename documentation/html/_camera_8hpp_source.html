394 FrustumPlaneState GetFrustumPlaneState(glm::vec3 point)
395 {
396 FrustumPlaneState res;
397
398 if (this->settings.isOrtho) {
399 glm::vec4 v = this->GetView() * glm::vec4(point, 1.0);
400 v = this->projection * v;
401
402 res.left = v.x >= frustumDataOrtho.min.x;
403 res.right = v.x <= frustumDataOrtho.max.x;
404 res.bottom = v.y >= frustumDataOrtho.min.y;
405 res.top = v.y <= frustumDataOrtho.max.y;
406 res.near = -v.z >= frustumDataOrtho.min.z;
407 res.far = v.z <= frustumDataOrtho.max.z;
408 }
409 else {
410 glm::vec4 v = glm::vec4(point, 1.0);
411 glm::vec4 calc = glm::vec4(0);
412
413 for (int i = 0; i < 4; i++) {
414 calc[i] = glm::dot(v, frustumDataPerspective.cols[i]);
415 }
416
417 res.left = -calc.w < calc.x ;
418 res.right = calc.x < calc.w;
419 res.bottom = -calc.w < calc.y;
420 res.top = calc.y < calc.w;
421 res.near = -calc.w < calc.z;
422 res.far = calc.z < calc.w;
423 }
424
425
426 return res;
427 }
