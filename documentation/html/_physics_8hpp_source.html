135 RaycastHit Raycast(GameObject* root, Camera* camera, double x, double y) {
136
137 if (camera != nullptr) {
138 glBindFramebuffer(GL_FRAMEBUFFER, camera->GetFrameBuffer());
139 glm::mat4 invProj = glm::inverse(camera->GetProjection());
140
141 glm::vec4 rayStartN = glm::vec4(x, y, -1.0f, 1.0f);
142 glm::vec4 rayEndN = glm::vec4(x, y, 0.0f, 1.0f);
143 glm::vec4 rayStartWorld = invProj * rayStartN;
144
145 rayStartWorld /= rayStartWorld.w;
146 glm::vec4 rayEndWorld = invProj * rayEndN;
147 rayEndWorld /= rayEndWorld.w;
148
149 rayStartWorld = camera->attachment->GetMatrixRecursive() * rayStartWorld;
150 rayEndWorld = camera->attachment->GetMatrixRecursive() * rayEndWorld;
151
152 glm::vec3 rayStart = glm::vec3(rayStartWorld.x, rayStartWorld.y, rayStartWorld.z);
153 glm::vec3 rayEnd = glm::vec3(rayEndWorld.x, rayEndWorld.y, rayEndWorld.z);
154 glm::vec3 rayDirection = glm::normalize(rayEnd - rayStart);
155
156 return Raycast(root, rayStart, rayDirection);
157 }
158
159 return RaycastHit(false, nullptr, glm::vec3(0), -1);
160 }

162 RaycastHit Raycast(GameObject* root, glm::vec3 origin, glm::vec3 dir) {
163 std::vector<RaycastObject*> ro = root->getComponentsByTypeRecursive<RaycastObject>();
164
165 RaycastHit res(false, nullptr, glm::vec3(0), -1);
166 for (size_t i = 0, max = ro.size(); i < max; i++) {
167 BoundingBoxCollider* bbc = ro[i]->attachment->getFirstComponentByType<BoundingBoxCollider>();
168 double val = -1;
169 int nearestVertex = -1;
170 if (bbc != nullptr) {
171 val = CollisionDetection::Ray_AABB(origin, dir, bbc);
172 }
173
174 if (val >= 0 && val < res.distance) {
175 res.distance = val;
176 res.hit = true;
177 res.obj = ro[i]->GetGameObject();
178 res.hitPosition = origin + (dir * (float)res.distance);
179 }
180 }
181
182 if (res.hit) {
183 Model* m = res.obj->getFirstComponentByType<Model>();
184 if (m != nullptr) {
185 std::vector<glm::vec3> pts = m->GetPoints();
186 unsigned int minPts = -1;
187 double minDist = std::numeric_limits<double>::max();
188 for (int i = 0, max = pts.size(); i < max; i++) {
189 double d = glm::distance(pts[i], res.hitPosition);
190 if (d < minDist) {
191 minPts = i;
192 minDist = d;
193 }
194 }
195 res.nearestVertex = minPts;
196 }
197 }
198 return res;
199 }
