68 void Render(Camera* camera, std::vector<Light*> lights, std::vector<Displayable*> elements, std::vector<WaterPhysics*> waterPhysics, bool mainRender, GameObject* root) {
69
70 glBindFramebuffer(GL_FRAMEBUFFER, mainRender ? 0 : camera->GetFrameBuffer());
71
72 //Update the camera Frustum
73 camera->UpdateFrustum();
74
75 //Test if the camera is in the water
76 bool isInWater = false;
77
78 if (waterPhysics.size() > 0 && camera->attachment->getFirstComponentByType<WaterAffected>() != nullptr) {
79 for (size_t wIndex = 0, wMax = waterPhysics.size(); wIndex < wMax && !isInWater; wIndex++) {
80 BoundingBoxCollider* bb = waterPhysics[wIndex]->attachment->getFirstComponentByType<BoundingBoxCollider>();
81 isInWater = CollisionDetection::Point_AABB(camera->GetPosition(), bb).collision;
82 }
83 }
84
85 //Clear the color and depth.
86 if (isInWater) {
87 glClearColor(0.0f, 0.66f, 0.8f, 1.0f);
88 }
89 else {
90 glClearColor(0.5f, 0.5f, 0.5f, 1.0f);
91 }
92
93 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
94
95 //Draw the elements
96 for (size_t indexElem = 0, nbElem = elements.size(); indexElem < nbElem; indexElem++) {
97 if (elements[indexElem]->IsAlwaysDraw()) {
98 Draw(camera, elements[indexElem], lights, isInWater, mainRender, root);
99 }
100 else
101 {
102 BoundingBoxCollider frustumCollider = elements[indexElem]->GetGameObject()->getFirstComponentByType<Model>()->GetFrustumCollider();
103 if (camera->IsInView(frustumCollider.GetMinOriented(), frustumCollider.GetMaxOriented())) {
104 //printf("In View\n");
105 Draw(camera, elements[indexElem], lights, isInWater, mainRender, root);
106 }
107 }
108 }
109
110 //Reset the buffer to the screen.
111 glBindFramebuffer(GL_FRAMEBUFFER, 0);
112 }

138
139 GameObject* go = element->GetGameObject();

141 return;

143
144 if (model == nullptr) {

147 bool updateLight = false;

149 IMaterial* renderMaterial = cam->GetRenderMaterial();

156

158 renderMaterial->SetDataGPU(go->GetMatrixRecursive(), cam->GetView(), cam->GetProjection(), cam->GetPosition(), waterFog, mainRender, caustics);
159 renderMaterial->SetLightGPU(lights);

180 glBindTexture(GL_TEXTURE_2D, wp->GetCaustics());

185

192
193
194 ModelInstanced* instanced = element->attachment->getFirstComponentByType<ModelInstanced>();

200 }
