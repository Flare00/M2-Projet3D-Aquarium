217 bool Fit(ICollider* collider) {
218 bool fit = false;
219 ICollider::Type type = collider->ColliderType();
220 if (type == ICollider::Type::BoundingBox) {
221 BoundingBoxCollider* bb = dynamic_cast<BoundingBoxCollider*>(collider);
222 if (bb != nullptr) {
223
224 if (Fit(bb->GetCenterOriented())) {
225 glm::vec3 pMin = bb->GetMinOriented();
226 glm::vec3 pMax = bb->GetMaxOriented();
227 glm::vec3 cMin = current.pos - glm::vec3(current.radius);
228 glm::vec3 cMax = current.pos + glm::vec3(current.radius);
229
230 fit = pMin.x >= cMin.x && pMax.x <= cMax.x &&
231 pMin.y >= cMin.y && pMax.y <= cMax.y &&
232 pMin.z >= cMin.z && pMax.z <= cMax.z;
233 }
234 }
235 }
236 else if (type == ICollider::Type::Sphere) {
237 SphereCollider* s = dynamic_cast<SphereCollider*>(collider);
238 if (s != nullptr) {
239 if (Fit(s->GetCenter())) {
240 glm::vec3 pMin = s->GetCenter() - glm::vec3(s->GetRadius());
241 glm::vec3 pMax = s->GetCenter() + glm::vec3(s->GetRadius());
242 glm::vec3 cMin = current.pos - glm::vec3(current.radius);
243 glm::vec3 cMax = current.pos + glm::vec3(current.radius);
244
245 fit = pMin.x >= cMin.x && pMax.x <= cMax.x &&
246 pMin.y >= cMin.y && pMax.y <= cMax.y &&
247 pMin.z >= cMin.z && pMax.z <= cMax.z;
