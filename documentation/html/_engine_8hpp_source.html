153 void MainLoop() {
154 double lastFrame = 0.0f;
155 double currentFrame = glfwGetTime();
156 double deltaTime = currentFrame - lastFrame;
157 lastFrame = currentFrame;
158 double sumDelta = 0.0;
159
160 do {
161 if (glfwGetKey(global.global_window, GLFW_KEY_ESCAPE) == GLFW_PRESS) {
162 global.close_asked = true;
163 }
164
165 double currentFrame = glfwGetTime();
166 deltaTime = currentFrame - lastFrame;
167 lastFrame = currentFrame;
168 sumDelta += deltaTime;
169
170 if (sumDelta >= 1.0 / (global.fps_limit)) {
171
172 std::string titre = "Moteur - ";
173 titre += std::to_string((int)((1.0 / sumDelta) + 0.1));
174 glfwSetWindowTitle(global.global_window, titre.c_str());
175
176 Loop(sumDelta);
177
178 glfwSwapBuffers(global.global_window);
179 glfwPollEvents();
180 sumDelta = 0.0f;
181 }
182
183 } while (!global.close_asked && glfwWindowShouldClose(global.global_window) == 0);
184 }

190 void Loop(double deltaT) {
191 if (activeScene >= 0 && activeScene < nbScene) {
192 reloadWait -= deltaT;
193 wireframeWait -= deltaT;
194 if (glfwGetKey(global.global_window, GLFW_KEY_R) == GLFW_PRESS && reloadWait < 0) {
195 scenes[activeScene].ReloadShaders();
196 reloadWait = 0.5;
197 }
198
199 if (glfwGetKey(global.global_window, GLFW_KEY_Z) == GLFW_PRESS && wireframeWait < 0) {
200 global.wireframe = !global.wireframe;
201 wireframeWait = 0.5;
202 }
203
204 scenes[activeScene].loop(deltaT);
205
206 GameObject* root = scenes[activeScene].GetRoot();
207 physics.Compute(deltaT, root);
208 graphics.Compute(root);
209 }
210 }
