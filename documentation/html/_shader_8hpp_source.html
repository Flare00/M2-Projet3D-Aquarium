140 int DefineOverride(std::vector<DataOverride> datas) {
141 int changes = 0;
142 for (DataOverride d : datas) {
143 bool change = false;
144 bool found = false;
145 for (int i = 0, max = this->previousDataOverride.size(); i < max && !found; i++) {
146 if (this->previousDataOverride[i].type == d.type) {
147 if (this->previousDataOverride[i].define.compare(d.define) == 0) {
148 found = true;
149 if (this->previousDataOverride[i].value.compare(d.value) != 0) {
150 change = true;
151 previousDataOverride[i].value = d.value;
152 }
153 }
154 }
155 }
156 if (!found) {
157 this->previousDataOverride.push_back(d);
158 }
159 if (change || !found) {
160 changes++;
161 switch (d.type) {
162 case VERTEX:
163 this->vertexData = ReplaceDefine(this->vertexData, d.define, d.value);
164 break;
165 case FRAGMENT:
166 this->fragmentData = ReplaceDefine(this->fragmentData, d.define, d.value);
167 break;
168 case TESS_CONTROL:
169 this->tesselationControlData = ReplaceDefine(this->tesselationControlData, d.define, d.value);
170 break;
171 case TESS_EVAL:
172 this->tesselationEvalData = ReplaceDefine(this->tesselationEvalData, d.define, d.value);
173 break;
174 case GEOMETRY:
175 this->geometryData = ReplaceDefine(this->geometryData, d.define, d.value);
176 break;
177 }
178 }
179 }
180 if (changes > 0) {
181 glDeleteProgram(program);
182 Compile();
183 }
184 return changes;
185 }

202 std::string ReplaceDefine(std::string data, std::string define, std::string value) {
203 std::string search = "#define " + define;
204 size_t found = data.find(search);
205 if (found >= 0) {
206 found += search.size();
207 size_t cursorStart = found + 1;
208 size_t cursorEnd = cursorStart;
209 char c = data[cursorStart];
210 bool end = false;
211 while (c != '\n' && c != ' ' && !end) {
212 cursorEnd++;
213 c = data[cursorEnd];
214 }
215 data = data.replace(cursorStart, cursorEnd - cursorStart, value).c_str();
216 }
217 //printf("DATA : %s\n", data.c_str());
218 return data;
219 }
