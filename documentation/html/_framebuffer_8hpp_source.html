135 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT3, GL_TEXTURE_2D, this->tex_reflection, 0);

139 glDrawBuffers(4, attachments);

222 void WriteTextureToFile(std::string name, int tex = 0) { //Tex : 0 = Color, 1 = Depth
223 if (w <= 0 || h <= 0)
224 return;
225
226 glBindTexture(GL_TEXTURE_2D, tex == 0 ? this->tex_color : this->tex_depth);
227
228 if (floating) {
229 GLsizei stride = 4 * this->w;
230 std::vector<GLfloat> datas(stride*this->h);
231 glGetTexImage(GL_TEXTURE_2D, 0, GL_RGBA, GL_FLOAT, datas.data());
232 std::vector <GLchar> datasChar;
233 for (int i = 0, max = datas.size(); i < max; i++) {
234 datasChar.push_back((char)(datas[i] * 255.0));
235 }
236 stbi_write_png(name.c_str(), w, h, 4, datasChar.data(), stride);
237 }
238 else {
239 GLsizei stride = 4 * this->w;
240
241 GLchar* datas = new GLchar[stride * h];
242 glGetTexImage(GL_TEXTURE_2D, 0, GL_RGBA, GL_UNSIGNED_BYTE, datas);
243
244
245 stbi_write_png(name.c_str(), w, h, 4, datas, stride);
246 }
247 }
